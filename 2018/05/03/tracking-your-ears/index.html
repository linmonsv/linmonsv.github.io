    cvtColor(frame, frameGray, CV\_BGR2GRAY);

    leftEarCascade.detectMultiScale(frameGray, leftEars, 1.1, 2, 0CV\_HAAR\_SCALE\_IMAGE, Size(30, 30) );
    
    // Detect right ear

    for(int i = 0; i < leftEars.size(); i++)
